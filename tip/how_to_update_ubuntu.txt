sudo apt-get update #패키지 인덱스 정보를 업데이트 
sudo apt-get upgrade # 설치된 패키지 업그레이드
sudo apt-get gist-upgrade # 의존성 검사하며 설치하기
sudo apt-get install "package name" # 패키지 설치
apt-get --reinstall install "package name" # 패키지 재설치
sudo apt-get remove "package name" # 패키지 삭제
sudo apt_get --purge remove "package name" # 패키지 삭제 -설정파일까지 지움
